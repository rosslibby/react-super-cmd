## Install

```bash
npm install --save react-super-cmd
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-super-cmd'
import 'react-super-cmd/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
